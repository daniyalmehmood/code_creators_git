public class YakRemover {
    public static String stringYak(String str) {
        StringBuilder str1 = new StringBuilder(str);
        for (int i = 0; i < str1.length()-1;) {
            if (str1.substring(i, i + 3).equals("yak")) {
                str1.delete(i, i + 3);
            }
            else {
                i++;
            }
        }
        return str1.toString();
    }

}
