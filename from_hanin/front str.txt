public class FrontRepeater {
    public static String repeatTimes(String str, int n) {
        if(n>=1 ) {
            int i;
          String m ="";
           String str1 =str.substring(0,3);
            for (i = 0; i< n; i++)
               m=str1 + m;
            return m;
        }
        return str;
    }

}
