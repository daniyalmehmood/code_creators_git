public class StringMatch {
    public static int countMatchingSubstrings(String a, String b) {
        int str = Math.min(a.length(), b.length());
       int count=0;
for (int i = 0; i<str-1; i++){
    String str1=a.substring(i,i+2);
    String str2=b.substring(i,i+2);
    if (str1.equals(str2)) {
        count++;
    }

}
        return count;
    }
}
