public class SumDouble {
    public static int SumDouble(int a, int b) {
        int sum = 0;
        if (a != b) {
            return sum = a + b; }
        if (a==b) {
            return sum = 2 * (a + b);
        }
    }


    public static void main(String[] args) {
        System.out.println(SumDouble(1, 2));
        System.out.println(SumDouble(3, 2));
        System.out.println(SumDouble(2, 2));
    }
}
