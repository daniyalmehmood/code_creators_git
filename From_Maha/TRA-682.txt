
public class notString {
    public static String checkNotString(String str){
          if (str.length() >= 3 && str.substring(0,3).equals("Not")) {
            return str ;

        }
        else {
            return ("Not " +str);
        }

    }
    public static void main(String[] args){
        System.out.println(checkNotString("Maha"));
    }
}
