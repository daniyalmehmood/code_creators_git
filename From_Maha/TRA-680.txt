ppublic class posNeg { 
    public static boolean testPosNeg(int a, int b, boolean negative){
        if ((a < 0 || b < 0) && !negative) {
            return true;
        }
       if ((a<0 || b<0) && negative==false) {
            return true;
        } else {
            return false;
        }
    }

    public static void main(String[] args) {
        System.out.println(testPosNeg(1, -1, false));
    }
}
