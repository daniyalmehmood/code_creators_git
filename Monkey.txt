public class Monkey {
    public static boolean monkeyTrouble ( boolean aSmile, boolean bSmile)
    {
        if (aSmile == true || bSmile == true) {
            System.out.println(" Monkey Trouble");
            return true;
        } else if (aSmile == false || bSmile == false) {
            System.out.println(" Monkey Trouble");
            return true;
        } else {
            System.out.println(" Monkey not in Trouble");
            return false;
        }
    }
}
