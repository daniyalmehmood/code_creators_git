public int stringMatch(String a, String b) {
        int len;
        int count = 0;

        if (a.length() > b.length()) {
            len = b.length();
        }
        else {
            len = a.length();
        }
        for (int i = 0; i < len-1; i++) {
            String asub = a.substring(i, i+2);
            String bsub = b.substring(i, i+2);

            if (asub.equals(bsub)) {
                count++;
            }
        }
        return count;
}
