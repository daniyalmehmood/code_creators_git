public String altPairs(String str) {
  String NewStr="";
  if (str.length()<=2){
    return str;
  }
for (int i=0 ; i<=str.length(); i=i+4 ){
    int last = i + 2;                           
    if (last > str.length()) {                    
      last = str.length();
}
  NewStr=NewStr+str.substring(i, last);

}  
return NewStr;
}
