public class IntMax {
    public static int intMax(int a, int b, int c) {
        if (a > b && a > c) {
            return a;
        } else if (b > a && b > c) {
            return b;
        } else if (c > a && c > b) {
            return c;
        } else if (a == b && a > c) {
            return a;
        } else if (a == c && a > b) {
            return c;
        } else {
            return b;
        }
    }
      public static void main (String[]args){
        System.out.println(intMax(1, 2, 3));
        System.out.println(intMax(1, 3, 2));
        System.out.println(intMax(3, 2, 1));
        System.out.println(intMax(1, 4, 4));
        System.out.println(intMax(4, 4, 0));
        System.out.println(intMax(1, 4, 4));
}
}
